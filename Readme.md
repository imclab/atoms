```js
var headers = [
  'number',
  'symbol',
  null,
  null,
  'period',
  'weight',
  'density',
  'melting-point',
  'boiling-point',
  'heating-point',
  'electronegativity',
  'abundance'
];

function extractBody() {
  var data = [];

  $('table.wikitable tbody tr').each(function(){
    if (!$(this).is(':visible')) return;

    var item = {};

    $('td', this).each(function(i){
      if (!headers[i]) return;

      // remove junk
      $('sup, span:hidden', this).remove();

      // have to do some manual logic here
      switch (headers[i]) {
        case 'density':
          item.density = value($('.sorttext', this).text());
          break;
        default:
          item[headers[i]] = value($(this).text());
          break;
      }
    });

    data.push(item);
  });

  function value(text) {
    text = text.trim();
    if ('â€“' == text) return undefined;
    if (!text) return undefined;
    return text;
  }

  return data;
}

console.log(JSON.stringify(extractBody(), null, 2));
```
